<script>

const photos=[
"photo1.jpg",
"photo2.jpg",
"photo3.jpg",
"photo4.jpg",
"photo5.jpg",
"photo6.jpg",
"photo7.jpg",
"photo8.jpg"
];

function startRoulette(){

let speed=70;
let count=0;

const roulette=setInterval(()=>{

const random=Math.floor(Math.random()*photos.length);

document.getElementById("photo").src=photos[random];

count++;

if(count>35){

clearInterval(roulette);

sparkleEffect();

}

},speed);

}

function sparkleEffect(){

for(let i=0;i<60;i++){

let s=document.createElement("div");

s.className="sparkle";

document.body.appendChild(s);

s.style.left=Math.random()*window.innerWidth+"px";
s.style.top=Math.random()*window.innerHeight+"px";

s.animate(

[
{transform:"scale(0)",opacity:1},
{transform:"scale(2)",opacity:0}
],

{
duration:800,
easing:"ease-out"
}

);

setTimeout(()=>{

s.remove();

},800);

}

}
